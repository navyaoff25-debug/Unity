switch

-> checks for each case

-> if any case is true then breaks

-> if no case is true default is executed

switch (expression) {

  case x:

  printf();

  break;

  case y:

  printf();

  break;

  default:

  printf()

}

eg:

int day = 4;

switch (day) {

  case 6:

  printf("Today is Saturday");

  break;

  case 7:

  printf("Today is Sunday");

  break;

  default:

  printf("Looking forward to the Weekend");

} //Looking forward to the Weekend

while loop

-> runs code until the condition fails

-> i++ is important without this the loop will never end

-> if condition is false, won't run atall

eg:

int i = 0;

while(i<5){

  print("%d\n", i );

  i++;

} // 0 1 2 3 4

int countdown = 3;

while (countdown > 0) {

  printf("%d\n", countdown);

  countdown--;

}

printf("Happy New Year!!\n"); // 0 1 2 Happy New Year!!

do/while loop

-> 1st runs once

-> then checks the condition

-> even though the condition is false, it is executed once

  printf("%d\n", i);

  i++;

}

while (i > 5); //0

for loop

-> when count of loops is known

syntax:

for(exp1; exp2; exp3) {

}

-> exp1 - executed before code execution

-> exp2 - defines condition

-> exp3 - executed after code execution

eg:

for (i=0; i<5, i++){

  printf("%d", i)

} // 0 1 2 3 4

nested loop

-> loop inside loop

int i, j;

for (i=0;i<=2;i++){

  printf("%d",i); // 2 times

  for (j=1;j<=3;j++){

  printf("%d",i); // 3 times // 2*3 - 6 times

  }

}

break

-> break the loop completely once the condition is passed

int i;

for (i = 0; i < 10; i++) {

  if (i == 4) {

  break;

  }

  printf("%d\n", i);

} // 0 1 2 3

continue

-> skips the round when condition is true, loop is continued

int i;

for (i = 0; i < 10; i++) {

  if (i == 4) {

  continue;

  }

  printf("%d\n", i);

} // 0 1 2 3 5 6 7 8 9
