#include <stdio.h>
int main() {
    int num1, num2;
    char operation;
    printf("Enter num1: ");
    scanf("%d", &num1);
    printf("Enter num2: ");
    scanf("%d", &num2);
    printf("Enter operation (+, -, *, /): ");
    scanf(" %c", &operation);
if (operation == '+'){
    printf("%d",num1+num2);
} else if (operation == '-') {
    printf("%d",num1-num2);
} else if (operation == '*') {
    printf("%d",num1*num2);
} else if (operation == '/') {
    if (num2 == 0){
        printf("Zero division error");
    } else {
        printf("%d",num1/num2);
    }
}
    return 0;
}
